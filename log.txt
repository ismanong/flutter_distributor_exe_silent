function IsCreateShortcut(): Boolean;
var
  j: Integer;
  isVerySilent: Boolean;
begin
  isVerySilent := True;
  for j := 1 to ParamCount do
    if CompareText(ParamStr(j), '--noicon=1') = 0 then
    begin
      isVerySilent := False;
      Break;
    end; 

  if isVerySilent then
    Log ('VerySilent')
  else
    Log ('not VerySilent');
  Result := isVerySilent;
end;
